zz_valyria_magic_fire_magic.002 = {
	type = character_event
	window = zz_valyria_magic_character_event
	title = zz_valyria_magic_fire_magic.002.t
	desc = zz_valyria_magic_fire_magic.002.desc
	theme = university 
	override_background = {
		reference = zz_fire_magic
	}
	left_portrait = root
	lower_center_portrait = {
		character = scope:secondary_recipient
	}
	right_portrait = scope:recipient

	immediate = {
		clear_variable_list = possible_armies
		remove_variable = selected_army
		scope:recipient = {
			every_army = {
				root = {
					add_to_variable_list = {
						name = possible_armies
						target = prev
					}
					if = {
						limit = {
							NOT = { has_variable = selected_army }
						}
						set_variable = { name = selected_army value = prev }
					}
				}
			}
		}
	}

	widget = {
		is_shown = {
			is_ai = no
		}
		gui = "event_window_army_selection"
		container = "custom_widgets_container"
	}
	option = {
		name = zz_valyria_magic_fire_magic.002.opt.a
		hidden_effect = {
			var:selected_army ?= { 
				save_scope_as = target
			}
			zz_valyria_magic_cast_effect = { 
				SKILL = fire_magic
				SPELL  = zz_valyria_magic_fire_magic_02
			}
		}
	}
	option = {
		name = zz_valyria_magic_fire_magic.002.opt.z
	}
}
